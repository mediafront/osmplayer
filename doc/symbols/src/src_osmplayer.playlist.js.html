1 /** The osmplayer namespace. */
  2 var osmplayer = osmplayer || {};

  4 /**
  5  * @constructor
  6  * @extends minplayer.display
  7  * @class This class creates the playlist functionality for the minplayer.
  8  *
  9  * @param {object} context The jQuery context.
 10  * @param {object} options This components options.
 11  */
 12 osmplayer.playlist = function(context, options) {
 13 
 14   // Derive from display
 15   minplayer.display.call(this, 'playlist', context, options);
 16 };
 17 
 18 /** Derive from minplayer.display. */
 19 osmplayer.playlist.prototype = new minplayer.display();
 20 
 21 /** Reset the constructor. */
 22 osmplayer.playlist.prototype.constructor = osmplayer.playlist;
 23 
 24 /**
 25  * @see minplayer.plugin#construct
 26  */
 27 osmplayer.playlist.prototype.construct = function() {
 28 
 29   // Make sure we provide default options...
 30   this.options = jQuery.extend({
 31     vertical: true,
 32     playlist: '',
 33     pageLimit: 10,
 34     autoNext: true,
 35     shuffle: false,
 36     loop: false,
 37     hysteresis: 40,
 38     scrollSpeed: 20,
 39     scrollMode: 'auto'
 40   }, this.options);
 41 
 42   // Call the minplayer plugin constructor.
 43   minplayer.display.prototype.construct.call(this);
 44 
 45   /** The nodes within this playlist. */
 46   this.nodes = [];
 47 
 48   // Current page.
 49   this.page = -1;
 50 
 51   // The total amount of nodes.
 52   this.totalItems = 0;
 53 
 54   // The current loaded item index.
 55   this.currentItem = -1;
 56 
 57   // The play playqueue.
 58   this.playqueue = [];
 59 
 60   // The playqueue position.
 61   this.playqueuepos = 0;
 62 
 63   // The current playlist.
 64   this.playlist = this.options.playlist;
 65 
 66   // Create the scroll bar.
 67   this.scroll = null;
 68   if (this.elements.scroll && (this.elements.scroll.length > 0)) {
 69 
 70     // Make this component orientation agnostic.
 71     var orient = {};
 72     orient.pos = (this.options.vertical) ? 'y' : 'x';
 73     orient.pagePos = (this.options.vertical ? 'pageY' : 'pageX');
 74     orient.offset = (this.options.vertical ? 'top' : 'left');
 75     orient.wrapperSize = (this.options.vertical ? 'wrapperH' : 'wrapperW');
 76     orient.minScroll = (this.options.vertical ? 'minScrollY' : 'minScrollX');
 77     orient.maxScroll = (this.options.vertical ? 'maxScrollY' : 'maxScrollX');
 78 
 79     // Setup the iScroll component.
 80     this.scroll = new iScroll(this.elements.scroll.eq(0)[0], {
 81       hScroll: !this.options.vertical,
 82       hScrollbar: !this.options.vertical,
 83       vScroll: this.options.vertical,
 84       vScrollbar: this.options.vertical,
 85       hideScrollbar: true
 86     });
 87 
 88     // Use autoScroll for non-touch devices.
 89     if ((this.options.scrollMode == 'auto') && !minplayer.hasTouch) {
 90 
 91       // Bind to the mouse events for autoscrolling.
 92       this.elements.list.bind('mousemove', (function(playlist) {
 93         return function(event) {
 94           event.preventDefault();
 95           playlist.mousePos = event[orient.pagePos];
 96           playlist.mousePos -= playlist.display.offset()[orient.offset];
 97         };
 98       })(this)).bind('mouseenter', (function(playlist) {
 99         return function(event) {
100           event.preventDefault();
101           playlist.scrolling = true;
102           var setScroll = function() {
103             if (playlist.scrolling) {
104               var scrollSize = playlist.scroll[orient.wrapperSize];
105               var scrollMid = (scrollSize / 2);
106               var delta = playlist.mousePos - scrollMid;
107               if (Math.abs(delta) > playlist.options.hysteresis) {
108                 var hyst = playlist.options.hysteresis;
109                 hyst *= (delta > 0) ? -1 : 0;
110                 delta = (playlist.options.scrollSpeed * (delta + hyst));
111                 delta /= scrollMid;
112                 var pos = playlist.scroll[orient.pos] - delta;
113                 var min = playlist.scroll[orient.minScroll] || 0;
114                 var max = playlist.scroll[orient.maxScroll];
115                 if (pos >= min) {
116                   playlist.scrollTo(min);
117                 }
118                 else if (pos <= max) {
119                   playlist.scrollTo(max);
120                 }
121                 else {
122                   playlist.scrollTo(delta, true);
123                 }
124               }
125 
126               // Set timeout to try again.
127               setTimeout(setScroll, 30);
128             }
129           };
130           setScroll();
131         };
132       })(this)).bind('mouseleave', (function(playlist) {
133         return function(event) {
134           event.preventDefault();
135           playlist.scrolling = false;
136         };
137       })(this));
138     }
139   }
140 
141   // Create the pager.
142   this.pager = this.create('pager', 'osmplayer');
143   this.pager.bind('nextPage', (function(playlist) {
144     return function(event) {
145       playlist.nextPage();
146     };
147   })(this));
148   this.pager.bind('prevPage', (function(playlist) {
149     return function(event) {
150       playlist.prevPage();
151     };
152   })(this));
153 
154   // Get the media.
155   if (this.options.autoNext) {
156     this.get('media', function(media) {
157       media.bind('ended', (function(playlist) {
158         return function(event) {
159           media.options.autoplay = true;
160           playlist.next();
161         };
162       })(this));
163     });
164   }
165 
166   // Load the "next" item.
167   this.next();
168 
169   // Say that we are ready.
170   this.ready();
171 };
172 
173 /**
174  * Wrapper around the scroll scrollTo method.
175  *
176  * @param {number} pos The position you would like to set the list.
177  * @param {boolean} relative If this is a relative position change.
178  */
179 osmplayer.playlist.prototype.scrollTo = function(pos, relative) {
180   if (this.scroll) {
181     this.scroll.options.hideScrollbar = false;
182     if (this.options.vertical) {
183       this.scroll.scrollTo(0, pos, 0, relative);
184     }
185     else {
186       this.scroll.scrollTo(pos, 0, 0, relative);
187     }
188     this.scroll.options.hideScrollbar = true;
189   }
190 };
191 
192 /**
193  * Refresh the scrollbar.
194  */
195 osmplayer.playlist.prototype.refresh = function() {
196   // Refresh the sizes.
197   if (this.scroll) {
198 
199     // Need to force the width of the list.
200     if (!this.options.vertical) {
201       var listSize = 0;
202       jQuery.each(this.elements.list.children(), function() {
203         listSize += jQuery(this).outerWidth();
204       });
205       this.elements.list.width(listSize);
206     }
207 
208     this.scroll.refresh();
209     this.scroll.scrollTo(0, 0, 200);
210   }
211 };

213 /**
214  * Sets the playlist.
215  *
216  * @param {object} playlist The playlist object.
217  * @param {integer} loadIndex The index of the item to load.
218  */
219 osmplayer.playlist.prototype.set = function(playlist, loadIndex) {
220 
221   // Check to make sure the playlist is an object.
222   if (typeof playlist !== 'object') {
223     this.trigger('error', 'Playlist must be an object to set');
224     return;
225   }
226 
227   // Check to make sure the playlist has correct format.
228   if (!playlist.hasOwnProperty('total_rows')) {
229     this.trigger('error', 'Unknown playlist format.');
230     return;
231   }
232 
233   // Make sure the playlist has some rows.
234   if (playlist.total_rows && playlist.nodes.length) {
235 
236     // Set the total rows.
237     this.totalItems = playlist.total_rows;
238     this.currentItem = 0;
239 
240     // Show or hide the next page if there is or is not a next page.
241     if (((this.page + 1) * this.options.pageLimit) >= this.totalItems) {
242       this.pager.nextPage.hide();
243     }
244     else {
245       this.pager.nextPage.show();
246     }
247 
248     var teaser = null;
249     var numNodes = playlist.nodes.length;
250     this.elements.list.empty();
251     this.nodes = [];
252 
253     // Iterate through all the nodes.
254     for (var index = 0; index < numNodes; index++) {
255 
256       // Create the teaser object.
257       teaser = this.create('teaser', 'osmplayer', this.elements.list);
258       teaser.setNode(playlist.nodes[index]);
259       teaser.bind('nodeLoad', (function(playlist, index) {
260         return function(event, data) {
261           playlist.loadItem(index);
262         };
263       })(this, index));
264 
265       // Add this to our nodes array.
266       this.nodes.push(teaser);
267 
268       // If the index is equal to the loadIndex.
269       if (loadIndex === index) {
270         this.loadItem(index);
271       }
272     }
273 
274     // Refresh the sizes.
275     this.refresh();
276 
277     // Trigger that the playlist has loaded.
278     this.trigger('playlistLoad', playlist);
279   }
280 
281   // Show that we are no longer busy.
282   if (this.elements.playlist_busy) {
283     this.elements.playlist_busy.hide();
284   }
285 };

287 /**
288  * Stores the current playlist state in the playqueue.
289  */
290 osmplayer.playlist.prototype.setQueue = function() {
291 
292   // Add this item to the playqueue.
293   this.playqueue.push({
294     page: this.page,
295     item: this.currentItem
296   });
297 
298   // Store the current playqueue position.
299   this.playqueuepos = this.playqueue.length;
300 };
301 
302 /**
303  * Loads the next item.
304  */
305 osmplayer.playlist.prototype.next = function() {
306   var item = 0, page = this.page;
307 
308   // See if we are at the front of the playqueue.
309   if (this.playqueuepos >= this.playqueue.length) {
310 
311     // If this is shuffle, then load a random item.
312     if (this.options.shuffle) {
313       item = Math.floor(Math.random() * this.totalItems);
314       page = Math.floor(item / this.options.pageLimit);
315       item = item % this.options.pageLimit;
316       this.load(page, item);
317     }
318     else {
319 
320       // Otherwise, increment the current item by one.
321       item = (this.currentItem + 1);
322       if (item >= this.nodes.length) {
323         this.load(page + 1, 0);
324       }
325       else {
326         this.loadItem(item);
327       }
328     }
329   }
330   else {
331 
332     // Load the next item in the playqueue.
333     this.playqueuepos = this.playqueuepos + 1;
334     var currentQueue = this.playqueue[this.playqueuepos];
335     this.load(currentQueue.page, currentQueue.item);
336   }
337 };
338 
339 /**
340  * Loads the previous item.
341  */
342 osmplayer.playlist.prototype.prev = function() {
343 
344   // Move back into the playqueue.
345   this.playqueuepos = this.playqueuepos - 1;
346   this.playqueuepos = (this.playqueuepos < 0) ? 0 : this.playqueuepos;
347   var currentQueue = this.playqueue[this.playqueuepos];
348   if (currentQueue) {
349     this.load(currentQueue.page, currentQueue.item);
350   }
351 };

353 /**
354  * Loads a playlist node.
355  *
356  * @param {number} index The index of the item you would like to load.
357  */
358 osmplayer.playlist.prototype.loadItem = function(index) {
359   if (index < this.nodes.length) {
360     this.setQueue();
361 
362     // Get the teaser at the current index and deselect it.
363     var teaser = this.nodes[this.currentItem];
364     teaser.select(false);
365     this.currentItem = index;
366 
367     // Get the new teaser and select it.
368     teaser = this.nodes[index];
369     teaser.select(true);
370     this.trigger('nodeLoad', teaser.node);
371   }
372 };

392 /**
393  * Loads a playlist.
394  *
395  * @param {integer} page The page to load.
396  * @param {integer} loadIndex The index of the item to load.
397  */
398 osmplayer.playlist.prototype.load = function(page, loadIndex) {
399 
400   // If the playlist and pages are the same, then no need to load.
401   if ((this.playlist == this.options.playlist) && (page == this.page)) {
402     this.loadItem(loadIndex);
403   }
404 
405   // Set the new playlist.
406   this.playlist = this.options.playlist;
407 
408   // Return if there aren't any playlists to play.
409   if (!this.playlist) {
410     return;
411   }
412 
413   // Determine if we need to loop.
414   var maxPages = Math.floor(this.totalItems / this.options.pageLimit);
415   if (page > maxPages) {
416     if (this.options.loop) {
417       page = 0;
418       loadIndex = 0;
419     }
420     else {
421       return;
422     }
423   }
424 
425   // Say that we are busy.
426   if (this.elements.playlist_busy) {
427     this.elements.playlist_busy.show();
428   }
429 
430   // Normalize the page.
431   page = page || 0;
432   page = (page < 0) ? 0 : page;
433 
434   // Set the queue.
435   this.setQueue();
436 
437   // Set the new page.
438   this.page = page;
439 
440   // Hide or show the page based on if we are on the first page.
441   if (this.page == 0) {
442     this.pager.prevPage.hide();
443   }
444   else {
445     this.pager.prevPage.show();
446   }
447 
448   // If the playlist is an object, then go ahead and set it.
449   if (typeof this.playlist == 'object') {
450     this.set(this.playlist, loadIndex);
451     if (this.playlist.endpoint) {
452       this.playlist = this.options.playlist = this.playlist.endpoint;
453     }
454     return;
455   }
456 
457   // Get the highest priority parser.
458   var parser = osmplayer.parser['default'];
459   for (var name in osmplayer.parser) {
460     if (osmplayer.parser.hasOwnProperty(name)) {
461       if (osmplayer.parser[name].valid(this.playlist)) {
462         if (osmplayer.parser[name].priority > parser.priority) {
463           parser = osmplayer.parser[name];
464         }
465       }
466     }
467   }
468 
469   // The start index.
470   var start = this.page * this.options.pageLimit;
471 
472   // Get the feed from the parser.
473   var feed = parser.getFeed(
474     this.playlist,
475     start,
476     this.options.pageLimit
477   );
478 
479   // Build our request.
480   var request = {
481     type: 'GET',
482     url: feed,
483     success: (function(playlist) {
484       return function(data) {
485         playlist.set(parser.parse(data), loadIndex);
486       };
487     })(this),
488     error: (function(playlist) {
489       return function(XMLHttpRequest, textStatus, errorThrown) {
490         if (playlist.elements.playlist_busy) {
491           playlist.elements.playlist_busy.hide();
492         }
493         playlist.trigger('error', textStatus);
494       }
495     })(this)
496   };
497 
498   // Set the data if applicable.
499   var dataType = '';
500   if (dataType = parser.getType()) {
501     request.dataType = dataType;
502   }
503 
504   // Perform an ajax callback.
505   jQuery.ajax(request);
506 };
