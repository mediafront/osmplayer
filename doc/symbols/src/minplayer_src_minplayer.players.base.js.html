111 /**
112  * Sets up a new media player.
113  */
114 minplayer.players.base.prototype.setupPlayer = function() {
115 
116   // Get the player display object.
117   if (!this.playerFound()) {
118 
119     // Add the new player.
120     this.addPlayer();
121   }
122 
123   // Get the player object...
124   this.player = this.getPlayer();
125 
126   // Toggle playing if they click.
127   minplayer.click(this.display, (function(player) {
128     return function() {
129       if (player.playing) {
130         player.pause();
131       }
132       else {
133         player.play();
134       }
135     };
136   })(this));
137 
138   // Bind to key events...
139   jQuery(document).bind('keydown', (function(player) {
140     return function(event) {
141       if (player.hasFocus) {
142         event.preventDefault();
143         switch (event.keyCode) {
144           case 32:  // SPACE
145           case 179: // GOOGLE play/pause button.
146             if (player.playing) {
147               player.pause();
148             }
149             else {
150               player.play();
151             }
152             break;
153           case 38:  // UP
154             player.setVolumeRelative(0.1);
155             break;
156           case 40:  // DOWN
157             player.setVolumeRelative(-0.1);
158             break;
159           case 37:  // LEFT
160           case 227: // GOOGLE TV REW
161             player.seekRelative(-0.05);
162             break;
163           case 39:  // RIGHT
164           case 228: // GOOGLE TV FW
165             player.seekRelative(0.05);
166             break;
167         }
168       }
169     };
170   })(this));
171 };

214 /**
215  * Resets all variables.
216  */
217 minplayer.players.base.prototype.reset = function() {
218 
219   // The duration of the player.
220   this.realDuration = 0;
221   this.duration = new minplayer.async();
222 
223   // The current play time of the player.
224   this.currentTime = new minplayer.async();
225 
226   // The amount of bytes loaded in the player.
227   this.bytesLoaded = new minplayer.async();
228 
229   // The total amount of bytes for the media.
230   this.bytesTotal = new minplayer.async();
231 
232   // The bytes that the download started with.
233   this.bytesStart = new minplayer.async();
234 
235   // The current volume of the player.
236   this.volume = new minplayer.async();
237 
238   // Reset focus.
239   this.hasFocus = false;
240 
241   // We are not playing.
242   this.playing = false;
243 
244   // We are not loading.
245   this.loading = false;
246 
247   // If we are loaded.
248   this.loaded = false;
249 
250   // Tell everyone else we reset.
251   this.trigger('pause', null, true);
252   this.trigger('waiting', null, true);
253   this.trigger('progress', {loaded: 0, total: 0, start: 0}, true);
254   this.trigger('timeupdate', {currentTime: 0, duration: 0}, true);
255 };

257 /**
258  * Called when the player is ready to recieve events and commands.
259  */
260 minplayer.players.base.prototype.onReady = function() {
261 
262   // Only continue if we are not already ready.
263   if (this.playerReady) {
264     return;
265   }
266 
267   // Set the ready flag.
268   this.playerReady = true;
269 
270   // Set the volume to the default.
271   this.setVolume(this.options.volume / 100);
272 
273   // Setup the progress interval.
274   this.loading = true;
275 
276   // Create a poll to get the progress.
277   this.poll('progress', (function(player) {
278     return function() {
279 
280       // Only do this if the play interval is set.
281       if (player.loading) {
282 
283         // Get the bytes loaded asynchronously.
284         player.getBytesLoaded(function(bytesLoaded) {
285 
286           // Get the bytes total asynchronously.
287           player.getBytesTotal(function(bytesTotal) {
288 
289             // Trigger an event about the progress.
290             if (bytesLoaded || bytesTotal) {
291 
292               // Get the bytes start, but don't require it.
293               var bytesStart = 0;
294               player.getBytesStart(function(val) {
295                 bytesStart = val;
296               });
297 
298               // Trigger a progress event.
299               player.trigger('progress', {
300                 loaded: bytesLoaded,
301                 total: bytesTotal,
302                 start: bytesStart
303               });
304 
305               // Say we are not longer loading if they are equal.
306               if (bytesLoaded >= bytesTotal) {
307                 player.loading = false;
308               }
309             }
310           });
311         });
312       }
313 
314       // Keep polling as long as its loading...
315       return player.loading;
316     };
317   })(this), 1000);
318 
319   // We are now ready.
320   this.ready();
321 
322   // Make sure the player is ready or errors will occur.
323   if (this.isReady()) {
324 
325     // Iterate through our ready queue.
326     for (var i in this.readyQueue) {
327       this.readyQueue[i].call(this);
328     }
329 
330     // Empty the ready queue.
331     this.readyQueue.length = 0;
332     this.readyQueue = [];
333 
334     if (!this.loaded) {
335 
336       // If we are still loading, then trigger that the load has started.
337       this.trigger('loadstart');
338     }
339   }
340   else {
341 
342     // Empty the ready queue.
343     this.readyQueue.length = 0;
344     this.readyQueue = [];
345   }
346 };

348 /**
349  * Parses a time value into seconds.
350  *
351  * @param string time
352  *   The time to parse to seconds.
353  *
354  * @returns {number}
355  *   The number of seconds this time represents.
356  */
357 minplayer.players.base.prototype.parseTime = function(time) {
358   var seconds = 0, minutes = 0, hours = 0;
359 
360   if (!time) {
361     return 0;
362   }
363 
364   // Convert to string if we need to.
365   if (typeof time != 'string') {
366     time = String(time);
367   }
368 
369   // Get the seconds.
370   seconds = time.match(/([0-9]+)s/i);
371   if (seconds) {
372     seconds = parseInt(seconds[1], 10);
373   }
374 
375   // Get the minutes.
376   minutes = time.match(/([0-9]+)m/i);
377   if (minutes) {
378     seconds += (parseInt(minutes[1], 10) * 60);
379   }
380 
381   // Get the hours.
382   hours = time.match(/([0-9]+)h/i);
383   if (hours) {
384     seconds += (parseInt(hours[1], 10) * 3600);
385   }
386 
387   // If no seconds were found, then just use the raw value.
388   if (!seconds) {
389     seconds = time;
390   }
391 
392   // Return the seconds from the time.
393   return Number(seconds);
394 };

396 /**
397  * Sets the start and stop points for the media.
398  *
399  * @return {number} The number of seconds we should seek.
400  */
401 minplayer.players.base.prototype.setStartStop = function() {
402   if (this.startTime) {
403     return this.startTime;
404   }
405 
406   this.startTime = 0;
407   this.offsetTime = this.parseTime(this.options.range.min);
408 
409   // First check the url for the seek time.
410   if (minplayer.urlVars) {
411     this.startTime = this.parseTime(minplayer.urlVars.seek);
412   }
413 
414   // Then check the options range parameter.
415   if (!this.startTime) {
416     this.startTime = this.offsetTime;
417   }
418 
419   // Get the stop time.
420   this.stopTime = this.options.range.max ? this.parseTime(this.options.range.max) : 0;
421 
422   // Calculate the range.
423   this.mediaRange = this.stopTime - this.offsetTime;
424   if (this.mediaRange < 0) {
425     this.mediaRange = 0;
426   }
427 
428   // Return the start time.
429   return this.startTime;
430 };

432 /**
433  * Should be called when the media is playing.
434  */
435 minplayer.players.base.prototype.onPlaying = function() {
436 
437   // Trigger an event that we are playing.
438   this.trigger('playing');
439 
440   // Say that this player has focus.
441   this.hasFocus = true;
442 
443   // Set the playInterval to true.
444   this.playing = true;
445   this.loaded = true;
446 
447   // Create a poll to get the timeupate.
448   this.poll('timeupdate', (function(player) {
449     return function() {
450 
451       // Only do this if the play interval is set.
452       if (player.playing) {
453 
454         // Get the current time asyncrhonously.
455         player.getCurrentTime(function(currentTime) {
456 
457           // Get the duration asynchronously.
458           player.getDuration(function(duration) {
459 
460             // Convert these to floats.
461             currentTime = parseFloat(currentTime);
462             duration = parseFloat(duration);
463 
464             // Trigger an event about the progress.
465             if (currentTime || duration) {
466 
467               // Trigger an update event.
468               player.trigger('timeupdate', {
469                 currentTime: currentTime,
470                 duration: duration
471               });
472             }
473           });
474         });
475       }
476 
477       // Keep polling as long as it is playing.
478       return player.playing;
479     };
480   })(this), 500);
481 };

513 /**
514  * Should be called when the media is done loading.
515  */
516 minplayer.players.base.prototype.onLoaded = function() {
517 
518   // See if we are loaded.
519   var isLoaded = this.loaded;
520 
521   // If we should autoplay, then just play now.
522   if (!this.loaded && this.options.autoplay) {
523     this.play();
524   }
525 
526   // We are now loaded.
527   this.loaded = true;
528 
529   // Trigger this event.
530   this.trigger('loadeddata');
531 
532   // See if they would like to seek.
533   if (!isLoaded) {
534     this.getDuration((function(player) {
535       return function(duration) {
536         if (player.startTime && (player.startTime < duration)) {
537           player.seek(player.startTime, null, true);
538           if (player.options.autoplay) {
539             player.play();
540           }
541         }
542       };
543     })(this));
544   }
545 };

698 /**
699  * Seeks to relative position.
700  *
701  * @param {number} pos Relative position.  -1 to 1 (percent), > 1 (seconds).
702  */
703 minplayer.players.base.prototype.seekRelative = function(pos) {
704 
705   // Get the current time asyncrhonously.
706   this.getCurrentTime((function(player) {
707     return function(currentTime) {
708 
709       // Get the duration asynchronously.
710       player.getDuration(function(duration) {
711 
712         // Only do this if we have a duration.
713         if (duration) {
714 
715           // Get the position.
716           var seekPos = 0;
717           if ((pos > -1) && (pos < 1)) {
718             seekPos = ((currentTime / duration) + parseFloat(pos)) * duration;
719           }
720           else {
721             seekPos = (currentTime + parseFloat(pos));
722           }
723 
724           // Set the seek value.
725           player.seek(seekPos);
726         }
727       });
728     };
729   })(this));
730 };

736  * @param {function} callback Called when it is done performing this operation.
737  */
738 minplayer.players.base.prototype.seek = function(pos, callback, noOffset) {
739   this.whenReady(function() {
740     pos = Number(pos);
741     if (!noOffset) {
742       pos += this.offsetTime;
743     }
744     this._seek(pos);
745     if (callback) {
746       callback.call(this);
747     }
748   });
749 };

861     callback(this.options.duration);
862   }
863   else {
864     var self = this;
865     this.getValue('_getDuration', 'duration', function(duration) {
866       self.setStartStop();
867       self.realDuration = duration;
868       callback(self.mediaRange ? self.mediaRange : duration);
869     });
870   }
871 };
