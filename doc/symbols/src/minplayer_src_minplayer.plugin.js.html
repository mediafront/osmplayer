1 /** The minplayer namespace. */
  2 minplayer = minplayer || {};
  3 
  4 /** Static array to keep track of all plugins. */
  5 minplayer.plugins = minplayer.plugins || {};
  6 
  7 /** Static array to keep track of queues. */
  8 minplayer.queue = minplayer.queue || [];
  9 
 10 /** Mutex lock to keep multiple triggers from occuring. */
 11 minplayer.lock = false;
 12 
 13 /**
 14  * @constructor
 15  * @class The base class for all plugins.
 16  *
 17  * @param {string} name The name of this plugin.
 18  * @param {object} context The jQuery context.
 19  * @param {object} options This components options.
 20  * @param {object} queue The event queue to pass events around.
 21  */
 22 minplayer.plugin = function(name, context, options, queue) {
 23 
 24   /** The name of this plugin. */
 25   this.name = name;
 26 
 27   /** The ready flag. */
 28   this.pluginReady = false;
 29 
 30   /** The options for this plugin. */
 31   this.options = options || {};
 32 
 33   /** The event queue. */
 34   this.queue = queue || {};
 35 
 36   /** Keep track of already triggered events. */
 37   this.triggered = {};
 38 
 39   /** Create a queue lock. */
 40   this.lock = false;
 41 
 42   // Only call the constructor if we have a context.
 43   if (context) {
 44 
 45     /** Keep track of the context. */
 46     this.context = jQuery(context);
 47 
 48     // Construct this plugin.
 49     this.construct();
 50   }
 51 };
 52 
 53 /**
 54  * The constructor which is called once the context is set.
 55  * Any class deriving from the plugin class should place all context
 56  * dependant functionality within this function instead of the standard
 57  * constructor function since it is called on object derivation as well
 58  * as object creation.
 59  */
 60 minplayer.plugin.prototype.construct = function() {
 61 
 62   // Adds this as a plugin.
 63   this.addPlugin();
 64 };
 65 
 66 /**
 67  * Destructor.
 68  */
 69 minplayer.plugin.prototype.destroy = function() {
 70 
 71   // Unbind all events.
 72   this.unbind();
 73 };

 75 /**
 76  * Creates a new plugin within this context.
 77  *
 78  * @param {string} name The name of the plugin you wish to create.
 79  * @param {object} base The base object for this plugin.
 80  * @param {object} context The context which you would like to create.
 81  * @return {object} The new plugin object.
 82  */
 83 minplayer.plugin.prototype.create = function(name, base, context) {
 84   var plugin = null;
 85 
 86   // Make sure we have a base object.
 87   base = base || 'minplayer';
 88   if (!window[base][name]) {
 89     base = 'minplayer';
 90   }
 91 
 92   // Make sure there is a context.
 93   context = context || this.display;
 94 
 95   // See if this plugin exists within this object.
 96   if (window[base][name]) {
 97 
 98     // Set the plugin.
 99     plugin = window[base][name];
100 
101     // See if a template version of the plugin exists.
102     if (plugin[this.options.template]) {
103 
104       plugin = plugin[this.options.template];
105     }
106 
107     // Make sure the plugin is a function.
108     if (typeof plugin !== 'function') {
109       plugin = window['minplayer'][name];
110     }
111 
112     // Make sure it is a function.
113     if (typeof plugin === 'function') {
114       return new plugin(context, this.options);
115     }
116   }
117 
118   return null;
119 };

121 /**
122  * Plugins should call this method when they are ready.
123  */
124 minplayer.plugin.prototype.ready = function() {
125 
126   // Keep this plugin from triggering multiple ready events.
127   if (!this.pluginReady) {
128 
129     // Set the ready flag.
130     this.pluginReady = true;
131 
132     // Now trigger that I am ready.
133     this.trigger('ready');
134 
135     // Check the queue.
136     this.checkQueue();
137   }
138 };

149 /**
150  * Adds a new plugin to this player.
151  *
152  * @param {string} name The name of this plugin.
153  * @param {object} plugin A new plugin object, derived from media.plugin.
154  */
155 minplayer.plugin.prototype.addPlugin = function(name, plugin) {
156   name = name || this.name;
157   plugin = plugin || this;
158 
159   // Make sure the plugin is valid.
160   if (plugin.isValid()) {
161 
162     // If the plugins for this instance do not exist.
163     if (!minplayer.plugins[this.options.id]) {
164 
165       // Initialize the plugins.
166       minplayer.plugins[this.options.id] = {};
167     }
168 
169     // Add this plugin.
170     minplayer.plugins[this.options.id][name] = plugin;
171 
172     // Now check the queue for this plugin.
173     this.checkQueue(plugin);
174   }
175 };

177 /**
178  * Create a polling timer.
179  *
180  * @param {function} callback The function to call when you poll.
181  * @param {integer} interval The interval you would like to poll.
182  */
183 minplayer.plugin.prototype.poll = function(callback, interval) {
184   setTimeout((function(context) {
185     return function callLater() {
186       if (callback.call(context)) {
187         setTimeout(callLater, interval);
188       }
189     };
190   })(this), interval);
191 };

193 /**
194  * Gets a plugin by name and calls callback when it is ready.
195  *
196  * @param {string} plugin The plugin of the plugin.
197  * @param {function} callback Called when the plugin is ready.
198  * @return {object} The plugin if no callback is provided.
199  */
200 minplayer.plugin.prototype.get = function(plugin, callback) {
201 
202   // If they pass just a callback, then return all plugins when ready.
203   if (typeof plugin === 'function') {
204     callback = plugin;
205     plugin = null;
206   }
207 
208   // Return the minplayer.get equivalent.
209   return minplayer.get.call(this, this.options.id, plugin, callback);
210 };

212 /**
213  * Check the queue and execute it.
214  *
215  * @param {object} plugin The plugin object to check the queue against.
216  */
217 minplayer.plugin.prototype.checkQueue = function(plugin) {
218 
219   // Initialize our variables.
220   var q = null, i = 0, check = false, newqueue = [];
221 
222   // Normalize the plugin variable.
223   plugin = plugin || this;
224 
225   // Set the lock.
226   minplayer.lock = true;
227 
228   // Iterate through all the queues.
229   var length = minplayer.queue.length;
230   for (i = 0; i < length; i++) {
231     if (minplayer.queue.hasOwnProperty(i)) {
232       // Get the queue.
233       q = minplayer.queue[i];
234 
235       // Now check to see if this queue is about us.
236       check = !q.id && !q.plugin;
237       check |= (q.plugin == plugin.name);
238       check &= (!q.id || (q.id == this.options.id));
239 
240       // If the check passes...
241       if (check) {
242         check = minplayer.bind.call(
243           q.context,
244           q.event,
245           this.options.id,
246           plugin.name,
247           q.callback
248         );
249       }
250 
251       // Add the queue back if it doesn't check out.
252       if (!check) {
253 
254         // Add this back to the queue.
255         newqueue.push(q);
256       }
257     }
258   }
259 
260   // Set the old queue to the new queue.
261   minplayer.queue = newqueue;
262 
263   // Release the lock.
264   minplayer.lock = false;
265 };

267 /**
268  * Trigger a media event.
269  *
270  * @param {string} type The event type.
271  * @param {object} data The event data object.
272  * @return {object} The plugin object.
273  */
274 minplayer.plugin.prototype.trigger = function(type, data) {
275   // Add this to our triggered array.
276   this.triggered[type] = data;
277 
278   // Check to make sure the queue for this type exists.
279   if (this.queue.hasOwnProperty(type)) {
280 
281     var i = 0, queue = {}, queuetype = this.queue[type];
282 
283     // Iterate through all the callbacks in this queue.
284     for (i in queuetype) {
285 
286       // Check to make sure the queue index exists.
287       if (queuetype.hasOwnProperty(i)) {
288 
289         // Setup the event object, and call the callback.
290         queue = queuetype[i];
291         queue.callback({target: this, data: queue.data}, data);
292       }
293     }
294   }
295 
296   // Return the plugin object.
297   return this;
298 };

300 /**
301  * Bind to a media event.
302  *
303  * @param {string} type The event type.
304  * @param {object} data The data to bind with the event.
305  * @param {function} fn The callback function.
306  * @return {object} The plugin object.
307  **/
308 minplayer.plugin.prototype.bind = function(type, data, fn) {
309 
310   // Allow the data to be the callback.
311   if (typeof data === 'function') {
312     fn = data;
313     data = null;
314   }
315 
316   // You must bind to a specific event and have a callback.
317   if (!type || !fn) {
318     return;
319   }
320 
321   // Initialize the queue for this type.
322   this.queue[type] = this.queue[type] || [];
323 
324   // Unbind any existing equivalent events.
325   this.unbind(type, fn);
326 
327   // Now add this event to the queue.
328   this.queue[type].push({
329     callback: fn,
330     data: data
331   });
332 
333   // Now see if this event has already been triggered.
334   if (this.triggered.hasOwnProperty(type)) {
335 
336     // Go ahead and trigger the event.
337     fn({target: this, data: data}, this.triggered[type]);
338   }
339 
340   // Return the plugin.
341   return this;
342 };

344 /**
345  * Unbind a media event.
346  *
347  * @param {string} type The event type.
348  * @param {function} fn The callback function.
349  * @return {object} The plugin object.
350  **/
351 minplayer.plugin.prototype.unbind = function(type, fn) {
352 
353   // If this is locked then try again after 10ms.
354   if (this.lock) {
355     setTimeout((function(plugin) {
356       return function() {
357         plugin.unbind(type, fn);
358       };
359     })(this), 10);
360   }
361 
362   // Set the lock.
363   this.lock = true;
364 
365   // Get the queue type.
366   var queuetype = this.queue.hasOwnProperty(type) ? this.queue[type] : null;
367 
368   if (!type) {
369     this.queue = {};
370   }
371   else if (!fn) {
372     this.queue[type] = [];
373   }
374   else if (queuetype) {
375     // Iterate through all the callbacks and search for equal callbacks.
376     var i = 0, queue = {};
377     for (i in queuetype) {
378       if (queuetype.hasOwnProperty(i)) {
379         if (queuetype[i].callback === fn) {
380           queue = this.queue[type].splice(1, 1);
381           delete queue;
382         }
383       }
384     }
385   }
386 
387   // Reset the lock.
388   this.lock = false;
389 
390   // Return the plugin.
391   return this;
392 };

394 /**
395  * Adds an item to the queue.
396  *
397  * @param {object} context The context which this is called within.
398  * @param {string} event The event to trigger on.
399  * @param {string} id The player ID.
400  * @param {string} plugin The name of the plugin.
401  * @param {function} callback Called when the event occurs.
402  */
403 minplayer.addQueue = function(context, event, id, plugin, callback) {
404 
405   // See if it is locked...
406   if (!minplayer.lock) {
407     minplayer.queue.push({
408       context: context,
409       id: id,
410       event: event,
411       plugin: plugin,
412       callback: callback
413     });
414   }
415   else {
416 
417     // If so, then try again after 10 milliseconds.
418     setTimeout(function() {
419       minplayer.addQueue(context, id, event, plugin, callback);
420     }, 10);
421   }
422 };

424 /**
425  * Binds an event to a plugin instance, and if it doesn't exist, then caches
426  * it for a later time.
427  *
428  * @param {string} event The event to trigger on.
429  * @param {string} id The player ID.
430  * @param {string} plugin The name of the plugin.
431  * @param {function} callback Called when the event occurs.
432  * @return {boolean} If the bind was successful.
433  * @this The object in context who called this method.
434  */
435 minplayer.bind = function(event, id, plugin, callback) {
436 
437   // If no callback exists, then just return false.
438   if (!callback) {
439     return false;
440   }
441 
442   // Get the plugins.
443   var plugins = minplayer.plugins;
444 
445   // Determine the selected plugins.
446   var selected = [];
447 
448   // If they provide id && plugin
449   if (id && plugin && plugins[id] && plugins[id][plugin]) {
450     selected.push(plugins[id][plugin]);
451   }
452 
453   // If they provide no id but a plugin.
454   else if (!id && plugin) {
455     for (var id in plugins) {
456       if (plugins[id].hasOwnProperty(plugin)) {
457         selected.push(plugins[id][plugin]);
458       }
459     }
460   }
461 
462   // If they provide an id but no plugin.
463   else if (id && !plugin && plugins[id]) {
464     for (var plugin in plugins[id]) {
465       selected.push(plugins[id][plugin]);
466     }
467   }
468 
469   // If they provide niether an id or a plugin.
470   else if (!id && !plugin) {
471     for (var id in plugins) {
472       for (var plugin in plugins[id]) {
473         selected.push(plugins[id][plugin]);
474       }
475     }
476   }
477 
478   // Iterate through the selected plugins and bind.
479   var i = selected.length;
480   while (i--) {
481     selected[i].bind(event, (function(context, plugin) {
482       return function(event) {
483         callback.call(context, event.target);
484       };
485     })(this, selected[i]));
486   }
487 
488   // See if there were any plugins selected.
489   if (selected.length == 0) {
490 
491     // If not, then add it to the queue to bind later.
492     minplayer.addQueue(this, event, id, plugin, callback);
493   }
494 
495   // Return that this wasn't handled.
496   return (selected.length > 0);
497 };

499 /**
500  * The main API for minPlayer.
501  *
502  * Provided that this function takes three parameters, there are 8 different
503  * ways to use this api.
504  *
505  *   id (0x100) - You want a specific player.
506  *   plugin (0x010) - You want a specific plugin.
507  *   callback (0x001) - You only want it when it is ready.
508  *
509  *   000 - You want all plugins from all players, ready or not.
510  *
511  *          var plugins = minplayer.get();
512  *
513  *   001 - You want all plugins from all players, but only when ready.
514  *
515  *          minplayer.get(function(plugin) {
516  *            // Code goes here.
517  *          });
518  *
519  *   010 - You want a specific plugin from all players, ready or not...
520  *
521  *          var medias = minplayer.get(null, 'media');
522  *
523  *   011 - You want a specific plugin from all players, but only when ready.
524  *
525  *          minplayer.get('player', function(player) {
526  *            // Code goes here.
527  *          });
528  *
529  *   100 - You want all plugins from a specific player, ready or not.
530  *
531  *          var plugins = minplayer.get('player_id');
532  *
533  *   101 - You want all plugins from a specific player, but only when ready.
534  *
535  *          minplayer.get('player_id', null, function(plugin) {
536  *            // Code goes here.
537  *          });
538  *
539  *   110 - You want a specific plugin from a specific player, ready or not.
540  *
541  *          var plugin = minplayer.get('player_id', 'media');
542  *
543  *   111 - You want a specific plugin from a specific player, only when ready.
544  *
545  *          minplayer.get('player_id', 'media', function(media) {
546  *            // Code goes here.
547  *          });
548  *
549  * @this The context in which this function was called.
550  * @param {string} id The ID of the widget to get the plugins from.
551  * @param {string} plugin The name of the plugin.
552  * @param {function} callback Called when the plugin is ready.
553  * @return {object} The plugin object if it is immediately available.
554  */
555 minplayer.get = function(id, plugin, callback) {
556 
557   // Normalize the arguments for a better interface.
558   if (typeof id === 'function') {
559     callback = id;
560     plugin = id = null;
561   }
562 
563   if (typeof plugin === 'function') {
564     callback = plugin;
565     plugin = id;
566     id = null;
567   }
568 
569   // Make sure the callback is a callback.
570   callback = (typeof callback === 'function') ? callback : null;
571 
572   // If a callback was provided, then just go ahead and bind.
573   if (callback) {
574     minplayer.bind.call(this, 'ready', id, plugin, callback);
575     return;
576   }
577 
578   // Get the plugins.
579   var plugins = minplayer.plugins;
580 
581   // 0x000
582   if (!id && !plugin && !callback) {
583     return plugins;
584   }
585   // 0x100
586   else if (id && !plugin && !callback) {
587     return plugins[id];
588   }
589   // 0x110
590   else if (id && plugin && !callback) {
591     return plugins[id][plugin];
592   }
593   // 0x010
594   else if (!id && plugin && !callback) {
595     var plugin_types = {};
596     for (var id in plugins) {
597       if (plugins[id].hasOwnProperty(plugin)) {
598         plugin_types[id] = plugins[id][plugin];
599       }
600     }
601     return plugin_types;
602   }
603 };
