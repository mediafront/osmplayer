 30         options.template = options.template || 'default';

 32           new osmplayer[options.template](jQuery(this), options);

 34         else {

 42 /**
 43  * @constructor
 44  * @extends minplayer
 45  * @class The main osmplayer class.
 46  *
 47  * <p><strong>Usage:</strong>
 48  * <pre><code>
 49  *
 50  *   // Create a media player.
 51  *   var player = jQuery("#player").osmplayer({
 52  *
 53  *   });
 54  *
 55  * </code></pre>
 56  * </p>
 57  *
 58  * @param {object} context The jQuery context.
 59  * @param {object} options This components options.
 60  */
 61 osmplayer = function(context, options) {
 62 
 63   // Derive from minplayer
 64   minplayer.call(this, context, options);

 67 /** Derive from minplayer. */

 73 /**
 74  * Creates a new plugin within this context.
 75  *
 76  * @param {string} name The name of the plugin you wish to create.
 77  * @param {object} base The base object for this plugin.
 78  * @param {object} context The context which you would like to create.
 79  * @return {object} The new plugin object.
 80  */
 81 osmplayer.prototype.create = function(name, base, context) {
 82   return minplayer.prototype.create.call(this, name, 'osmplayer', context);
 83 };

 85 /**
 86  * @see minplayer.plugin.construct
 87  */
 88 osmplayer.prototype.construct = function() {
 89 
 90   // Make sure we provide default options...
 91   this.options = jQuery.extend({
 92     playlist: '',
 93     node: {},
 94     link: 'http://www.mediafront.org'
 95   }, this.options);
 96 
 97   // Provide a default logo if one isn't provided.
 98   if (!this.options.logo) {
 99     this.options.logo = 'http://mediafront.org/assets/osmplayer/logo.png';
100   }
101 
102   // Call the minplayer display constructor.
103   minplayer.prototype.construct.call(this);
104 
105   // We need to cleanup the player when it has been destroyed.
106   jQuery(this.display).bind('playerdestroyed', (function(player) {
107     return function(element) {
108       if (element === player.display.eq(0)[0]) {
109         for (var plugin in minplayer.plugins[player.options.id]) {
110           for (var index in minplayer.plugins[player.options.id][plugin]) {
111             minplayer.plugins[player.options.id][plugin][index].destroy();
112             delete minplayer.plugins[player.options.id][plugin][index];
113           }
114           minplayer.plugins[player.options.id][plugin].length = 0;
115         }
116         delete minplayer.plugins[player.options.id];
117         minplayer.plugins[player.options.id] = null;
118       }
119     };
120   })(this));
121 
122   /** The play queue and index. */
123   this.playQueue = [];
124   this.playIndex = 0;
125   this.hasPlaylist = false;
126 
127   /** The playlist for this media player. */
128   this.create('playlist', 'osmplayer');
129 
130   /** Get the playlist or any other playlist that connects. */
131   this.get('playlist', function(playlist) {
132     this.hasPlaylist = true;
133     playlist.ubind(this.uuid + ':nodeLoad', (function(player) {
134       return function(event, data) {
135         player.loadNode(data);
136       };
137     })(this));
138   });
139 
140   // Play each media sequentially...
141   this.get('media', (function(player) {
142     return function(media) {
143       media.ubind(player.uuid + ':ended', function() {
144         player.options.autoplay = true;
145         player.playNext();
146       });
147     };
148   })(this));
149 
150   // Load the node if one is provided.
151   if (this.options.node) {
152     this.loadNode(this.options.node);
153   }
154 };

165 /**
166  * The load node function.
167  *
168  * @param {object} node A media node object.
169  */
170 osmplayer.prototype.loadNode = function(node) {
171   if (node && node.mediafiles) {
172 
173     // Load the media files.
174     var media = node.mediafiles.media;
175     if (media) {
176       this.playQueue.length = 0;
177       this.playQueue = [];
178       this.playIndex = 0;
179       var file = null;
180       var types = [];
181 
182       // For mobile devices, we should only show the main media.
183       if (minplayer.isAndroid || minplayer.isIDevice) {
184         types = ['media'];
185       }
186       else {
187         types = ['intro', 'commercial', 'prereel', 'media', 'postreel'];
188       }
189 
190       // Iterate through the types.
191       jQuery.each(types, (function(player) {
192         return function(key, type) {
193           if (file = player.addToQueue(media[type])) {
194             file.queueType = type;
195           }
196         };
197       })(this));
198     }
199     else {
200 
201       // Add a class to the display to let themes handle this.
202       this.display.addClass('nomedia');
203     }
204 
205     // Load the preview image.
206     osmplayer.getImage(node.mediafiles, 'preview', (function(player) {
207       return function(image) {
208         player.options.preview = image.path;
209         if (player.playLoader) {
210           player.playLoader.enabled = true;
211           player.playLoader.loadPreview();
212         }
213       };
214     })(this));
215 
216     // Play the next media
217     this.playNext();
218   }
219 };
220 
221 /**
222  * Adds a file to the play queue.
223  *
224  * @param {object} file The file to add to the queue.
225  * @return {object} The file that was added to the queue.
226  */
227 osmplayer.prototype.addToQueue = function(file) {
228   if (file = minplayer.getMediaFile(file)) {
229     this.playQueue.push(file);
230   }
231   return file;
232 };
233 
234 /**
235  * Plays the next media file in the queue.
236  */
237 osmplayer.prototype.playNext = function() {
238   if (this.playQueue.length > this.playIndex) {
239     this.load(this.playQueue[this.playIndex]);
240     this.playIndex++;
241   }
242   else if (this.options.repeat) {
243     this.playIndex = 0;
244     this.playNext();
245   }
246   else if (this.playQueue.length > 0) {
247 
248     // If we have a playlist, let them handle what to do next.
249     if (this.hasPlaylist && this.options.autoNext) {
250       this.trigger('player_ended');
251     }
252     else {
253       // If there is no playlist, and no repeat, we will
254       // just seek to the beginning and pause.
255       this.options.autoplay = false;
256       this.playIndex = 0;
257       this.playNext();
258     }
259   }
260   else if (this.media) {
261     // Stop the player and unload.
262     this.media.stop();
263   }
264 };

266 /**
267  * Returns an image provided image array.
268  *
269  * @param {object} mediafiles The mediafiles to search within.
270  * @param {string} type The type of image to look for.
271  * @param {function} callback Called when the image is retrieved.
272  */
273 osmplayer.getImage = function(mediafiles, type, callback) {
274 
275   var image = '';
276   var images = mediafiles.image;
277   if (images) {
278 
279     // If the image type exists, then just use that one...
280     if (images[type]) {
281       image = images[type];
282     }
283     // Or try the original image...
284     else if (images['image']) {
285       image = images['image'];
286     }
287     // Otherwise, just try ANY image...
288     else {
289 
290       // Or, just pick the first one available.
291       for (type in images) {
292         if (images.hasOwnProperty(type)) {
293           image = images[type];
294           break;
295         }
296       }
297     }
298   }
299 
300   // If the image exists, then callback with that image.
301   if (image) {
302     callback(new minplayer.file(image));
303   }
304   else {
305     // Get the image from the media player...
306     var mediaFile = minplayer.getMediaFile(mediafiles.media.media);
307     if (mediaFile) {
308       var player = minplayer.players[mediaFile.player];
309       if (player && (typeof player.getImage === 'function')) {
310         player.getImage(mediaFile, type, function(src) {
311           callback(new minplayer.file(src));
312         });
313       }
314     }
315   }
316 };
