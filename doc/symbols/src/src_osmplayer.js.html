  2 if (!jQuery.fn.osmplayer) {
  3 
  4   /**
  5    * @constructor
  6    *
  7    * Define a jQuery osmplayer prototype.
  8    *
  9    * @param {object} options The options for this jQuery prototype.
 10    * @return {Array} jQuery object.
 11    */
 12   jQuery.fn.osmplayer = function(options) {
 13     return jQuery(this).each(function() {
 14       options = options || {};
 15       options.id = options.id || jQuery(this).attr('id') || Math.random();
 16       if (!minplayer.plugins[options.id]) {
 17         options.template = options.template || 'default';
 18         if (osmplayer[options.template]) {
 19           new osmplayer[options.template](jQuery(this), options);
 20         }
 21         else {
 22           new osmplayer(jQuery(this), options);
 23         }
 24       }
 25     });
 26   };
 27 }

 29 /**
 30  * @constructor
 31  * @extends minplayer
 32  * @class The main osmplayer class.
 33  *
 34  * <p><strong>Usage:</strong>
 35  * <pre><code>
 36  *
 37  *   // Create a media player.

 39  *

 42  * </code></pre>
 43  * </p>
 44  *
 45  * @param {object} context The jQuery context.
 46  * @param {object} options This components options.

 48 osmplayer = function(context, options) {
 49 
 50   // Derive from minplayer
 51   minplayer.call(this, context, options);
 52 };

 54 /** Derive from minplayer. */
 55 osmplayer.prototype = new minplayer();
 56 
 57 /** Reset the constructor. */
 58 osmplayer.prototype.constructor = osmplayer;

 60 /**
 61  * Creates a new plugin within this context.
 62  *
 63  * @param {string} name The name of the plugin you wish to create.
 64  * @param {object} base The base object for this plugin.
 65  * @param {object} context The context which you would like to create.
 66  * @return {object} The new plugin object.
 67  */
 68 osmplayer.prototype.create = function(name, base, context) {
 69   return minplayer.prototype.create.call(this, name, 'osmplayer', context);
 70 };

 72 /**
 73  * @see minplayer.plugin.construct
 74  */
 75 osmplayer.prototype.construct = function() {
 76 
 77   // Make sure we provide default options...
 78   this.options = jQuery.extend({
 79     playlist: '',
 80     node: {},
 81     swfplayer: 'minplayer/flash/minplayer.swf',
 82     logo: 'logo.png',
 83     link: 'http://www.mediafront.org'
 84   }, this.options);
 85 
 86   // Call the minplayer display constructor.
 87   minplayer.prototype.construct.call(this);
 88 
 89   /** The play queue and index. */
 90   this.playQueue = [];
 91   this.playIndex = 0;
 92 
 93   /** The playlist for this media player. */
 94   this.create('playlist', 'osmplayer');
 95 
 96   /** Get the playlist or any other playlist that connects. */
 97   this.get('playlist', function(playlist) {
 98     playlist.bind('nodeLoad', (function(player) {
 99       return function(event, data) {
100         player.loadNode(data);
101       };
102     })(this));
103   });
104 
105   // Load the node if one is provided.
106   if (this.options.node) {
107     this.loadNode(this.options.node);
108   }
109 };

120 /**
121  * The load node function.
122  *
123  * @param {object} node A media node object.
124  */
125 osmplayer.prototype.loadNode = function(node) {
126   if (node && node.mediafiles) {
127 
128     // Load the media files.
129     var media = node.mediafiles.media;
130     if (media) {
131       this.playQueue.length = 0;
132       this.playQueue = [];
133       this.playIndex = 0;
134       this.addToQueue(media.intro);
135       this.addToQueue(media.commercial);
136       this.addToQueue(media.prereel);
137       this.addToQueue(media.media);
138       this.addToQueue(media.postreel);
139     }
140 
141     // Load the preview image.
142     this.options.preview = osmplayer.getImage(node.mediafiles.image, 'preview');
143 
144     if (this.playLoader) {
145       this.playLoader.loadPreview();
146     }
147 
148     // Play the next media
149     this.playNext();
150   }
151 };

153 /**
154  * Adds a file to the play queue.
155  *
156  * @param {object} file The file to add to the queue.
157  */
158 osmplayer.prototype.addToQueue = function(file) {
159   if (file) {
160     this.playQueue.push(this.getFile(file));
161   }
162 };
163 
164 /**
165  * Returns a valid media file for this browser.
166  *
167  * @param {object} file The file object.
168  * @return {object} The best media file.
169  */
170 osmplayer.prototype.getFile = function(file) {
171   if (file) {
172     var type = typeof file;
173     if (((type === 'object') || (type === 'array')) && file[0]) {
174       file = this.getBestMedia(file);
175     }
176   }
177   return file;
178 };
179 
180 /**
181  * Returns the media file with the lowest weight value provided an array of
182  * media files.
183  *
184  * @param {object} files The media files to play.
185  * @return {object} The best media file.
186  */
187 osmplayer.prototype.getBestMedia = function(files) {
188   var mFile = null;
189   var i = files.length;
190   while (i--) {
191     var tempFile = new minplayer.file(files[i]);
192     if (!mFile || (tempFile.priority > mFile.priority)) {
193       mFile = tempFile;
194     }
195   }
196   return mFile;
197 };
198 
199 /**
200  * Plays the next media file in the queue.
201  */
202 osmplayer.prototype.playNext = function() {
203   if (this.playQueue.length > this.playIndex) {
204     this.load(this.playQueue[this.playIndex]);
205     this.playIndex++;
206   }
207   else if (this.options.repeat) {
208     this.playIndex = 0;
209     this.playNext();
210   }
211   else {
212     // If there is no playlist, and no repeat, we will
213     // just seek to the beginning and pause.
214     this.options.autostart = false;
215     this.playIndex = 0;
216     this.playNext();
217   }
218 };

220 /**
221  * Returns an image provided image array.
222  *
223  * @param {object} images The images to search for.
224  * @param {string} type The type of image to look for.
225  * @return {object} The best image match.
226  */
227 osmplayer.getImage = function(images, type) {
228   var image = '';
229 
230   if (images) {
231 
232     // If the image type exists, then just use that one...
233     if (images[type]) {
234       image = images[type];
235     }
236     // Or try the original image...
237     else if (images['image']) {
238       image = images['image'];
239     }
240     // Otherwise, just try ANY image...
241     else {
242 
243       // Or, just pick the first one available.
244       for (type in images) {
245         if (images.hasOwnProperty(type)) {
246           image = images[type];
247           break;
248         }
249       }
250     }
251   }
252 
253   // Return the image path.
254   return (typeof image === 'string') ? image : image.path;
255 };
