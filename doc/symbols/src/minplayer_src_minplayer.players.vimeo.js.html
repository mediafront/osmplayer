 36 /**
 37  * @see minplayer.players.base#canPlay
 38  * @return {boolean} If this player can play this media type.
 39  */
 40 minplayer.players.vimeo.canPlay = function(file) {
 41 
 42   // Check for the mimetype for vimeo.
 43   if (file.mimetype === 'video/vimeo') {
 44     return true;
 45   }
 46 
 47   // If the path is a vimeo path, then return true.
 48   return (file.path.search(/^http(s)?\:\/\/(www\.)?vimeo\.com/i) === 0);
 49 };

 69 /**
 70  * Return the ID for a provided media file.
 71  *
 72  * @param {object} file A {@link minplayer.file} object.
 73  * @return {string} The ID for the provided media.
 74  */
 75 minplayer.players.vimeo.getMediaId = function(file) {
 76   var regex = /^http[s]?\:\/\/(www\.)?vimeo\.com\/(\?v\=)?([0-9]+)/i;
 77   if (file.path.search(regex) === 0) {
 78     return file.path.match(regex)[3];
 79   }
 80   else {
 81     return file.path;
 82   }
 83 };

 94 /**
 95  * @see minplayer.players.base#create
 96  * @return {object} The media player entity.
 97  */
 98 minplayer.players.vimeo.prototype.create = function() {
 99   minplayer.players.base.prototype.create.call(this);
100 
101   // Insert the Vimeo Froogaloop player.
102   var tag = document.createElement('script');
103   tag.src = 'http://a.vimeocdn.com/js/froogaloop2.min.js';
104   var firstScriptTag = document.getElementsByTagName('script')[0];
105   firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);
106 
107   // Create the iframe for this player.
108   var iframe = document.createElement('iframe');
109   iframe.setAttribute('id', this.options.id + '-player');
110   iframe.setAttribute('type', 'text/html');
111   iframe.setAttribute('width', '100%');
112   iframe.setAttribute('height', '100%');
113   iframe.setAttribute('frameborder', '0');
114 
115   // Get the source.
116   var src = 'http://player.vimeo.com/video/';
117   src += this.mediaFile.id + '?';
118 
119   // Add the parameters to the src.
120   src += jQuery.param({
121     'wmode': 'opaque',
122     'api': 1,
123     'player_id': this.options.id + '-player',
124     'title': 0,
125     'byline': 0,
126     'portrait': 0,
127     'autoplay': this.options.autoplay,
128     'loop': this.options.loop
129   });
130 
131   // Set the source of the iframe.
132   iframe.setAttribute('src', src);
133 
134   // Now register this player when the froogaloop code is loaded.
135   this.poll((function(player) {
136     return function() {
137       if (window.Froogaloop) {
138         player.player = window.Froogaloop(iframe);
139         var playerTimeout = 0;
140         player.player.addEvent('ready', function() {
141           clearTimeout(playerTimeout);
142           player.onReady();
143           player.onError('');
144         });
145         playerTimeout = setTimeout(function() {
146           player.onReady();
147           player.onError('Unable to play video.');
148         }, 2000);
149       }
150       return !window.Froogaloop;
151     };
152   })(this), 200);
153 
154   // Trigger that the load has started.
155   this.trigger('loadstart');
156 
157   // Return the player.
158   return iframe;
159 };

161 /**
162  * @see minplayer.players.base#onReady
163  */
164 minplayer.players.vimeo.prototype.onReady = function(player_id) {
165 
166   // Add the other listeners.
167   this.player.addEvent('loadProgress', (function(player) {
168     return function(progress) {
169       player.duration.set(parseFloat(progress.duration));
170       player.bytesLoaded.set(progress.bytesLoaded);
171       player.bytesTotal.set(progress.bytesTotal);
172     };
173   })(this));
174 
175   this.player.addEvent('playProgress', (function(player) {
176     return function(progress) {
177       player.duration.set(parseFloat(progress.duration));
178       player.currentTime.set(parseFloat(progress.seconds));
179     };
180   })(this));
181 
182   this.player.addEvent('play', (function(player) {
183     return function() {
184       player.onPlaying();
185     };
186   })(this));
187 
188   this.player.addEvent('pause', (function(player) {
189     return function() {
190       player.onPaused();
191     };
192   })(this));
193 
194   this.player.addEvent('finish', (function(player) {
195     return function() {
196       player.onComplete();
197     };
198   })(this));
199 
200   minplayer.players.base.prototype.onReady.call(this);
201   this.onLoaded();
202 };
